	SYMBOL	=	[^\r\n\f \t]+

		( css:rhs )
	S_RHS		=	->|→
		( css:rarrow,	translation:RARROW )
	E_RHS		=	\r\n|[\r\n\f]
		( css:linefeed,	translation:NEW_LINE )
	EPSILON		=	epsilon|ε
		( css:epsilon,	translation:SYMBOL_EPSILON )
